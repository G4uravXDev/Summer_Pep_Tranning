/**
 * Problem Name : Protect the Island
 * Link : https://codeforces.com/problemset/problem/1320/C
 * Approach : Sweep-Line with Segment Tree (Lazy Propagation)
 */

import java.io.*;
import java.util.*;

public class Main {
    static final int MAX_VAL = 1000000;
    static long[] tree = new long[4 * MAX_VAL + 5];
    static long[] lazy = new long[4 * MAX_VAL + 5];

    static class Weapon implements Comparable<Weapon> {
        int a, cost;
        public Weapon(int a, int cost) {
            this.a = a;
            this.cost = cost;
        }
        @Override
        public int compareTo(Weapon o) {
            return Integer.compare(this.a, o.a);
        }
    }

    static class Monster implements Comparable<Monster> {
        int x, y, z;
        public Monster(int x, int y, int z) {
            this.x = x;
            this.y = y;
            this.z = z;
        }
        @Override
        public int compareTo(Monster o) {
            return Integer.compare(this.x, o.x);
        }
    }

    static void build(int node, int start, int end, long[] initialValues) {
        if (start == end) {
            tree[node] = initialValues[start];
            return;
        }
        int mid = (start + end) >> 1;
        build(2 * node, start, mid, initialValues);
        build(2 * node + 1, mid + 1, end, initialValues);
        tree[node] = Math.max(tree[2 * node], tree[2 * node + 1]);
    }

    static void push(int node) {
        if (lazy[node] != 0) {
            lazy[2 * node] += lazy[node];
            tree[2 * node] += lazy[node];
            lazy[2 * node + 1] += lazy[node];
            tree[2 * node + 1] += lazy[node];
            lazy[node] = 0;
        }
    }

    static void update(int node, int start, int end, int l, int r, long val) {
        if (l > end || r < start) return;
        if (l <= start && end <= r) {
            tree[node] += val;
            lazy[node] += val;
            return;
        }
        push(node);
        int mid = (start + end) >> 1;
        update(2 * node, start, mid, l, r, val);
        update(2 * node + 1, mid + 1, end, l, r, val);
        tree[node] = Math.max(tree[2 * node], tree[2 * node + 1]);
    }

    static long query(int node, int start, int end, int l, int r) {
        if (l > end || r < start) return -1000000000000000000L;
        if (l <= start && end <= r) {
            return tree[node];
        }
        push(node);
        int mid = (start + end) >> 1;
        return Math.max(query(2 * node, start, mid, l, r), query(2 * node + 1, mid + 1, end, l, r));
    }

    public static void main(String[] args) {
        FastScanner sc = new FastScanner(System.in);
        int n = sc.nextInt();
        if (n == -1) return;
        int m = sc.nextInt();
        int p = sc.nextInt();

        Weapon[] weapons = new Weapon[n];
        for (int i = 0; i < n; i++) {
            weapons[i] = new Weapon(sc.nextInt(), sc.nextInt());
        }

        long[] min_armor_cost = new long[MAX_VAL + 5];
        Arrays.fill(min_armor_cost, 1000000000000000L);
        for (int i = 0; i < m; i++) {
            int b = sc.nextInt();
            long cb = sc.nextLong();
            min_armor_cost[b] = Math.min(min_armor_cost[b], cb);
        }

        Monster[] monsters = new Monster[p];
        for (int i = 0; i < p; i++) {
            monsters[i] = new Monster(sc.nextInt(), sc.nextInt(), sc.nextInt());
        }

        Arrays.sort(weapons);
        Arrays.sort(monsters);

        long[] initialValues = new long[MAX_VAL + 5];
        for (int i = 1; i <= MAX_VAL; i++) {
            initialValues[i] = -min_armor_cost[i];
        }

        build(1, 1, MAX_VAL, initialValues);

        long ans = -1000000000000000000L;
        int monster_ptr = 0;

        for (int i = 0; i < n; i++) {
            while (monster_ptr < p && monsters[monster_ptr].x < weapons[i].a) {
                int def = monsters[monster_ptr].y;
                int reward = monsters[monster_ptr].z;
                if (def + 1 <= MAX_VAL) {
                    update(1, 1, MAX_VAL, def + 1, MAX_VAL, reward);
                }
                monster_ptr++;
            }
            long current_max_armor_profit = query(1, 1, MAX_VAL, 1, MAX_VAL);
            long profit = current_max_armor_profit - weapons[i].cost;
            ans = Math.max(ans, profit);
        }

        System.out.println(ans);
    }

    static class FastScanner {
        private InputStream stream;
        private byte[] buf = new byte[1024 * 1024];
        private int head = 0;
        private int tail = 0;

        public FastScanner(InputStream stream) {
            this.stream = stream;
        }

        private int read() {
            if (head >= tail) {
                head = 0;
                try {
                    tail = stream.read(buf, 0, buf.length);
                } catch (IOException e) {
                    throw new RuntimeException(e);
                }
                if (tail <= 0) return -1;
            }
            return buf[head++];
        }

        public int nextInt() {
            int c = read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = read();
            }
            boolean neg = false;
            if (c == '-') {
                neg = true;
                c = read();
            }
            int res = 0;
            while (c > 32) {
                if (c >= '0' && c <= '9') {
                    res = res * 10 + c - '0';
                }
                c = read();
            }
            return neg ? -res : res;
        }

        public long nextLong() {
            int c = read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = read();
            }
            boolean neg = false;
            if (c == '-') {
                neg = true;
                c = read();
            }
            long res = 0;
            while (c > 32) {
                if (c >= '0' && c <= '9') {
                    res = res * 10 + c - '0';
                }
                c = read();
            }
            return neg ? -res : res;
        }
    }
}
