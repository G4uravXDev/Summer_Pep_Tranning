/**
 * Problem Name : ATM Transaction
 * Link : https://www.hackerrank.com/challenges/minimum-time-required/problem
 * Approach : Binary Search on Answer
 */

import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        long k = sc.nextLong();

        long[] a = new long[n];
        long maxVal = 0;
        for (int i = 0; i < n; i++) {
            a[i] = sc.nextLong();
            maxVal = Math.max(maxVal, a[i]);
        }

        long low = 1;
        long high = maxVal + k + 5;

        while (low <= high) {
            long mid = low + (high - low) / 2;

            if (check(a, k, mid)) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }

        System.out.println(high);
    }

    static boolean check(long[] a, long k, long x) {
        long need = 0;
        for (long val : a) {
            need += Math.max(0, x - val);
            if (need > k) return true;
        }
        return need > k;
    }
}
