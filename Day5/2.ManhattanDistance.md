/**
 * Problem Name : Manhattan Distances
 * Link : https://www.geeksforgeeks.org/dsa/sum-manhattan-distances-pairs-points/
 * Approach : Mathematical Observation
 */


import java.util.*;

class Solution {
    public long sumOfManhattanDistances(int[] x, int[] y) {
        int n = x.length;

        long totalSum = 0;

        for (int i = 0; i < n; i++) {
            long dx = 0;
            long dy = 0;

            for (int j = 0; j < n; j++) {
                dx += Math.abs(x[i] - x[j]);
                dy += Math.abs(y[i] - y[j]);
            }

            totalSum += dx + dy;
        }

        return totalSum;
    }
}
