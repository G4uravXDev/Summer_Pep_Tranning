/**
 * Problem Name : The Rival Team
 * Link : https://leetcode.com/problems/subarray-sum-equals-k/description/
 * Approach : Sliding Window
 */


import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int q = sc.nextInt();
        while (q-- > 0) {
            int n = sc.nextInt();
            long k = sc.nextLong();
            long[] a = new long[n];
            for (int i = 0; i < n; i++) {
                a[i] = sc.nextLong();
            }

            long ans = 0;
            long currentSum = 0;
            int left = 0;

            for (int right = 0; right < n; right++) {
                currentSum += a[right];

                while (currentSum > k) {
                    currentSum -= a[left];
                    left++;
                }

                ans += (right - left + 1);
            }

            System.out.println(ans);
        }
    }
}
