/**
 * Problem Name : Overlapping Intervals
 * Link : https://www.geeksforgeeks.org/problems/overlapping-intervals--170633/1
 * Approach : Sorting + 
 */



class Solution {
    public ArrayList<ArrayList<Integer>> mergeOverlap(int[][] arr) {
        Arrays.sort(arr, (a, b) -> Integer.compare(a[0], b[0]));

        ArrayList<ArrayList<Integer>> ans = new ArrayList<>();

        for (int[] interval : arr) {

            if (ans.isEmpty() || ans.get(ans.size() - 1).get(1) < interval[0]) {
                ArrayList<Integer> temp = new ArrayList<>();
                temp.add(interval[0]);
                temp.add(interval[1]);
                ans.add(temp);
            }
            else {
                ArrayList<Integer> last = ans.get(ans.size() - 1);
                last.set(1, Math.max(last.get(1), interval[1]));
            }
        }

        return ans;
    }
}
