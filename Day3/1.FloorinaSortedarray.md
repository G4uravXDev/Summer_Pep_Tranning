/**
 * Problem Name : Floor in a Sorted Array
 * Link : https://www.geeksforgeeks.org/problems/floor-in-a-sorted-array1017/1
 * Approach : Binary Search Algorithm
 */

class Solution {
    static int findFloor(long arr[], int n, long x) {
        int low = 0;
        int high = n - 1;
        int ans = -1;

        while (low <= high) {
            int mid = low + (high - low) / 2;

            if (arr[mid] == x) {
                return mid;
            } else if (arr[mid] < x) {
                ans = mid;
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }

        return ans;
    }
}
