/**
 * Problem Name : Ropes
 * Link : https://codeforces.com/edu/course/2/lesson/6/2/practice/contest/283932/problem/B
 * Approach : Binary Search on Answer
 */


import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int k = sc.nextInt();
        
        int[] a = new int[n];
        double maxLen = 0;
        for (int i = 0; i < n; i++) {
            a[i] = sc.nextInt();
            if (a[i] > maxLen) {
                maxLen = a[i];
            }
        }
        
        double low = 0;
        double high = maxLen;
        
        for (int iter = 0; iter < 100; iter++) {
            double mid = low + (high - low) / 2;
            
            long pieces = 0;
            for (int i = 0; i < n; i++) {
                pieces += (long)(a[i] / mid);
                if (pieces >= k) {
                    break;
                }
            }
            
            if (pieces >= k) {
                low = mid;
            } else {
                high = mid;
            }
        }
        
        System.out.printf(Locale.US, "%.7f\n", low);
    }
}
