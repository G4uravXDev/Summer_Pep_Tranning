/**
 * Problem Name : First Negative in Window of Size K
 * Link : https://www.geeksforgeeks.org/problems/first-negative-integer-in-every-window-of-size-k3345/1
 * Approach : Sliding Window with Deque
 */


import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) return;
        int n = sc.nextInt();
        int k = sc.nextInt();
        long[] a = new long[n];
        for (int i = 0; i < n; i++) {
            a[i] = sc.nextLong();
        }

        Deque<Integer> dq = new LinkedList<>();
        List<Long> ans = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            if (!dq.isEmpty() && dq.peekFirst() < i - k + 1) {
                dq.pollFirst();
            }

            if (a[i] < 0) {
                dq.addLast(i);
            }

            if (i >= k - 1) {
                if (!dq.isEmpty()) {
                    ans.add(a[dq.peekFirst()]);
                } else {
                    ans.add(0L);
                }
            }
        }

        for (int i = 0; i < ans.size(); i++) {
            System.out.print(ans.get(i) + (i == ans.size() - 1 ? "" : " "));
        }
        System.out.println();
    }
}
