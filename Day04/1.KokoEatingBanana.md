```python
# Problem Name : Koko Eating Banana
# Link : https://leetcode.com/problems/koko-eating-bananas/
# Approach : Binary Search on Answer

import math

class Solution:
    def minEatingSpeed(self, piles: list[int], h: int) -> int:
        low = 1
        high = 10**9
        ans = high

        while low <= high:
            mid = low + (high - low) // 2

            hours = 0
            for pile in piles:
                hours += math.ceil(pile / mid)

            if hours <= h:
                ans = mid
                high = mid - 1
            else:
                low = mid + 1

        return ans
```
