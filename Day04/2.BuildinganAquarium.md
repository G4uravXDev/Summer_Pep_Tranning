```python
# Problem Name : Building an Aquarium
# Link : https://codeforces.com/contest/1873/problem/E
# Approach : Binary Search on Answer

import sys

def check(a, k, x):
    need = 0
    for val in a:
        need += max(0, x - val)
        if need > k:
            return True
    return need > k

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    
    n = int(data[0])
    k = int(data[1])
    
    a = [int(x) for x in data[2:n+2]]
    max_val = max(a)
    
    low = 1
    high = max_val + k + 5
    
    while low <= high:
        mid = low + (high - low) // 2
        
        if check(a, k, mid):
            low = mid + 1
        else:
            high = mid - 1
            
    print(high)

if __name__ == '__main__':
    main()
```
