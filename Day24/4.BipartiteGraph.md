# Problem Name: Bipartite Graph
# Link : https://www.geeksforgeeks.org/problems/bipartite-graph/1
# Topic : Bipartite Graph


from collections import deque
from typing import List


class Solution:
    def isBipartite(self, V : int = None, adj : List[List[int]] = None) -> bool:
        if adj is None:
            adj = V
            V = len(adj)
        elif V is None:
            V = len(adj)
        
        color = [-1] * V
        for i in range(V):
            if color[i] == -1:
                if not self.bfs(i, color, adj):
                    return False
        return True
    
    def bfs(self, start_node: int, color: List[int], adj: List[List[int]]) -> bool:
        q = deque([start_node])
        color[start_node] = 0
        
        while q:
            node = q.popleft()
            
            for neighbor in adj[node]:
                if color[neighbor] == -1:
                    color[neighbor] = 1 - color[node]
                    q.append(neighbor)
                elif color[neighbor] == color[node]:
                    return False
        return True
