# Problem Name: Detect Cycle in an Undirected Graph
# Link : https://www.geeksforgeeks.org/problems/detect-cycle-in-an-undirected-graph/1
# Topic : Cycle in an Undirected Graph(BFS)

from collections import deque
from typing import List


class Solution:
    def isCycle(self, V: int = None, adj: List[List[int]] = None) -> bool:
        if adj is None:
            adj = V
            V = len(adj)
        elif V is None:
            V = len(adj)
            
        visited = [False] * V
        for i in range(V):
            if not visited[i]:
                if self.bfs(i, visited, adj):
                    return True
        return False
    
    def bfs(self, start_node: int, visited: List[bool], adj: List[List[int]]) -> bool:
        q = deque([(start_node, -1)]) # Store (node, parent)
        visited[start_node] = True
        
        while q:
            node, parent = q.popleft()
            
            for neighbor in adj[node]:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    q.append((neighbor, node))
                elif neighbor != parent:
                    return True
        return False
