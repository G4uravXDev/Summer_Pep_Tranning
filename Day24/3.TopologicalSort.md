# Problem Name: Topological Sort
# Link : https://www.geeksforgeeks.org/problems/topological-sort/1
# Topic : Topological Sort(Kahn's Algo)

from collections import deque
from typing import List


class Solution:
    def topologicalSort(self, V: int = None, adj: List[List[int]] = None) -> List[int]:
        if adj is None:
            adj = V
            V = len(adj)
        elif V is None:
            V = len(adj)
            
        in_degree = [0] * V
        for i in range(V):
            for neighbor in adj[i]:
                in_degree[neighbor] += 1
        
        q = deque()
        for i in range(V):
            if in_degree[i] == 0:
                q.append(i)
        
        result = []
        while q:
            node = q.popleft()
            result.append(node)
            
            for neighbor in adj[node]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    q.append(neighbor)
        
        return result
