```python
# Problem Name: Detect Cycle in a Directed Graph
# Link : https://www.geeksforgeeks.org/problems/detect-cycle-in-a-directed-graph/1
# Topic : Cycle in a Directed Graph(DFS)

from typing import List

class Solution:
    def isCycle(self, V: int = None, adj: List[List[int]] = None) -> bool:
        if adj is None:
            adj = V
            V = len(adj)
        elif V is None:
            V = len(adj)
            
        # 0: unvisited, 1: visiting (in current recursion stack), 2: visited (finished processing)
        state = [0] * V 
        
        for i in range(V):
            if state[i] == 0:
                if self.dfs(i, state, adj):
                    return True
        return False
    
    def dfs(self, node: int, state: List[int], adj: List[List[int]]) -> bool:
        state[node] = 1  # Mark as visiting
        
        for neighbor in adj[node]:
            if state[neighbor] == 1:  # Found a back edge to a node currently in recursion stack
                return True
            if state[neighbor] == 0:  # Unvisited neighbor
                if self.dfs(neighbor, state, adj):
                    return True
        
        state[node] = 2  # Mark as fully visited (finished processing)
        return False
```
