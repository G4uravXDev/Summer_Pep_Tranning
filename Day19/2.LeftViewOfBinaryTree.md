```python
# Problem Name: Left View Of Binary Tree
# Link : https://www.geeksforgeeks.org/problems/left-view-of-binary-tree/1
# Topic : Tree Traversal

from collections import deque
class Solution:
    def LeftView(self, root):
        if not root:
            return []
        
        q = deque([root])
        ans = []
        
        while q:
            level_size = len(q)
            ans.append(q[0].data)

            for _ in range(level_size):
                node = q.popleft()
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
        
        return ans
```
