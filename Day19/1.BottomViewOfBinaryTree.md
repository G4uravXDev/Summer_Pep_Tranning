# Problem Name: Bottom View Of Binary Tree
# Link : https://www.geeksforgeeks.org/problems/bottom-view-of-binary-tree/1
# Topic : Tree Traversal

from collections import deque

class Solution:
    def bottomView(self, root):
        if not root:
            return []

        d = {}
        q = deque([(root, 0)])

        while q:
            node, hd = q.popleft()
            d[hd] = node.data

            if node.left:
                q.append((node.left, hd - 1))
            if node.right:
                q.append((node.right, hd + 1))

        ans = []
        for hd in sorted(d):
            ans.append(d[hd])

        return ans
