```python
# Problem Name: Binary Tree Inorder Traversal
# Link : https://www.geeksforgeeks.org/problems/inorder-traversal/1
# Topic : Tree Traversal

# class Node:
#     def __init__(self, val):
#         self.data = val
#         self.left = None
#         self.right = None

class Solution:
    def InOrder(self, root):
        res = []
        
        def solve(node):
            if node is None:
                return
            solve(node.left)
            res.append(node.data)
            solve(node.right)
            
        solve(root)
        return res
```
