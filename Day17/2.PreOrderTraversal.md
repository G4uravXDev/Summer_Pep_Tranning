```python
# Problem Name: Binary Tree Preorder Traversal
# Link : https://www.geeksforgeeks.org/problems/preorder-traversal/1
# Topic : Tree Traversal

# class Node:
#     def __init__(self, val):
#         self.data = val
#         self.left = None
#         self.right = None

class Solution:
    def preorder(self, root):
        res = []

        def solve(node):
            if node is None:
                return
            res.append(node.data)
            solve(node.left)
            solve(node.right)

        solve(root)
        return res
```
