```python
# Problem Name : Ropes
# Link : https://codeforces.com/edu/course/2/lesson/6/2/practice/contest/283932/problem/B
# Approach : Binary Search on Answer

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    k = int(data[1])
    
    a = [int(x) for x in data[2:n+2]]
    max_len = max(a)
    
    low = 0.0
    high = float(max_len)
    
    for _ in range(100):
        mid = low + (high - low) / 2
        
        pieces = 0
        for val in a:
            pieces += int(val / mid)
            if pieces >= k:
                break
                
        if pieces >= k:
            low = mid
        else:
            high = mid
            
    print(f"{low:.7f}")

if __name__ == '__main__':
    main()
```
