```python
# Problem Name : Factory Machines
# Link : https://cses.fi/problemset/task/1620/
# Approach : Binary search on Answer

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    t = int(data[1])
    
    k = [int(x) for x in data[2:n+2]]
    
    low = 1
    high = 10**18
    ans = high
    
    while low <= high:
        mid = low + (high - low) // 2
        
        products = 0
        for machine in k:
            products += mid // machine
            if products >= t:
                break
                
        if products >= t:
            ans = mid
            high = mid - 1
        else:
            low = mid + 1
            
    print(ans)

if __name__ == '__main__':
    main()
```
