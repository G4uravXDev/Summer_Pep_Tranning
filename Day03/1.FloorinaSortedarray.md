```python
# Problem Name : Floor in a Sorted Array
# Link : https://www.geeksforgeeks.org/problems/floor-in-a-sorted-array1017/1
# Approach : Binary Search Algorithm

class Solution:
    def findFloor(self, arr: list[int], n: int, x: int) -> int:
        low = 0
        high = n - 1
        ans = -1

        while low <= high:
            mid = low + (high - low) // 2

            if arr[mid] == x:
                return mid
            elif arr[mid] < x:
                ans = mid
                low = mid + 1
            else:
                high = mid - 1

        return ans
```
