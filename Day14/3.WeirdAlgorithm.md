# Problem Name: Weird Algorithm
# Link : https://cses.fi/problemset/task/1068
# Topic : Recursion

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    
    ans = [str(n)]
    while n != 1:
        if n % 2 == 0:
            n = n // 2
        else:
            n = 3 * n + 1
        ans.append(str(n))
        
    print(" ".join(ans))

if __name__ == '__main__':
    main()
