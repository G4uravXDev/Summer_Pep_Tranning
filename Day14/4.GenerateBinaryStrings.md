```python
# Problem Name: Generate All Binary Strings
# Link : https://www.geeksforgeeks.org/problems/generate-all-binary-strings/1
# Topic : Recursion

class Solution:
    def binstr(self, n: int) -> list[str]:
        ans = []
        
        def solve(curr):
            if len(curr) == n:
                ans.append(curr)
                return
            solve(curr + "0")
            solve(curr + "1")
            
        solve("")
        return ans
```
