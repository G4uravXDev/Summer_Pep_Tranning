```python
# Problem Name: kth Largest Element in BST
# Link : https://www.geeksforgeeks.org/problems/kth-largest-element-in-bst/1
# Topic : Binary Search Tree


class Solution:
    def kthLargest(self, root, k):
        if root is None:
            return None

        if root.data == k:
            return root.data

        if root.data < k:
            return self.kthLargest(root.right, k)
        else:
            return self.kthLargest(root.left, k)
```
