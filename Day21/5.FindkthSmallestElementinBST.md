```python
# Problem Name: Find K-th Smallest Element in BST
# Link : https://www.geeksforgeeks.org/problems/find-k-th-smallest-element-in-bst/1
# Topic : Binary Search Tree


class Solution:
    def kthSmallest(self, root, k):
        if root is None:
            return None

        if root.data == k:
            return root

        if root.data < k:
            return self.kthSmallest(root.right, k)
        else:
            return self.kthSmallest(root.left, k)
```
