```python
# Problem Name: Implementing Ceil in BST
# Link : https://www.geeksforgeeks.org/problems/implementing-ceil-in-bst/1
# Topic : Binary Search Tree


class Solution:
    def ceil(self, root, x):
        if root is None:
            return None

        if root.data == x:
            return root

        if root.data < x:
            return self.ceil(root.right, x)
        else:
            return self.ceil(root.left, x)
```
