```python
# Problem Name: Lowest Common Ancestor in a Binary Tree
# Link : https://www.geeksforgeeks.org/problems/lowest-common-ancestor-in-a-binary-tree/1
# Topic : Tree Traversal


class Solution:
    def lca(self, root, n1, n2):
        if root is None:
            return None

        if root.data == n1 or root.data == n2:
            return root

        left_lca = self.lca(root.left, n1, n2)
        right_lca = self.lca(root.right, n1, n2)

        if left_lca and right_lca:
            return root

        return left_lca if left_lca else right_lca
```
