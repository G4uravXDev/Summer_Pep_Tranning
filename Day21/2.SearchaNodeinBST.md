```python
# Problem Name: Search a Node in a BST
# Link : https://www.geeksforgeeks.org/problems/search-a-node-in-bst/1
# Topic : Binary Search Tree


class Solution:
    def searchBST(self, root, x):
        if root is None or root.data == x:
            return root
        
        if x < root.data:
            return self.searchBST(root.left, x)
        else:
            return self.searchBST(root.right, x)
```
