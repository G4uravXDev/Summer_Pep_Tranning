/**
 * Problem Name : Search In LinkedList
 * Link : http://geeksforgeeks.org/problems/search-in-linked-list-1664434326/1
 * Approach : Linked List
 */



int searchInLinkedList(Node head, int x)
{
    Node current = head;

    while (current != null)
    {
        if (current.data == x)
            return 1;

        current = current.next;
    }

    return 0;
}
