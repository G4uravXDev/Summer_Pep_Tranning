```python
# Problem Name : LinkedList Cycle
# Link : https://leetcode.com/problems/linked-list-cycle/description/
# Approach : Linked List

class Solution:
    def hasCycle(self, head: ListNode) -> bool:
        slow = head
        fast = head

        while fast is not None and fast.next is not None:
            slow = slow.next
            fast = fast.next.next

            if slow == fast:
                return True

        return False
```
