/**
 * Problem Name : Middle Of The LinkedList
 * Link : https://leetcode.com/problems/middle-of-the-linked-list/description/
 * Approach : Linked List
 */



class Solution {
    public ListNode middleNode(ListNode head) {
        ListNode slow = head;
        ListNode fast = head;

        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }

        return slow;
    }
}
