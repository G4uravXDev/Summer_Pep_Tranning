```python
# Problem Name : ATM Transaction (Minimum Time Required)
# Link : https://www.hackerrank.com/challenges/minimum-time-required/problem
# Approach : Binary Search on Answer

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    goal = int(data[1])
    machines = [int(x) for x in data[2:n+2]]
    
    machines.sort()
    
    low = 1
    high = machines[0] * goal
    ans = high
    
    while low <= high:
        mid = (low + high) // 2
        
        produced = 0
        for m in machines:
            produced += mid // m
            
        if produced >= goal:
            ans = mid
            high = mid - 1
        else:
            low = mid + 1
            
    print(ans)

if __name__ == '__main__':
    main()
```
