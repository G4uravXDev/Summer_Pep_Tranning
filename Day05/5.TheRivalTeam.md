```python
# Problem Name : The Rival Team
# Link : https://leetcode.com/problems/subarray-sum-equals-k/description/
# Approach : Sliding Window

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    q = int(data[0])
    
    idx = 1
    out = []
    for _ in range(q):
        n = int(data[idx])
        k = int(data[idx+1])
        idx += 2
        
        a = []
        for _ in range(n):
            a.append(int(data[idx]))
            idx += 1
            
        ans = 0
        current_sum = 0
        left = 0
        
        for right in range(n):
            current_sum += a[right]
            
            while current_sum > k:
                current_sum -= a[left]
                left += 1
                
            ans += (right - left + 1)
        out.append(str(ans))
        
    print("\n".join(out))

if __name__ == '__main__':
    main()
```
