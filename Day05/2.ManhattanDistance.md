```python
# Problem Name : Manhattan Distances
# Link : https://www.geeksforgeeks.org/dsa/sum-manhattan-distances-pairs-points/
# Approach : Mathematical Observation (O(N log N))

class Solution:
    def sumOfManhattanDistances(self, x: list[int], y: list[int]) -> int:
        n = len(x)
        if n <= 1:
            return 0
        
        def get_1d_sum(coords):
            coords.sort()
            total_sum = 0
            for i in range(n):
                total_sum += coords[i] * i - coords[i] * (n - 1 - i)
            return total_sum
            
        return 2 * (get_1d_sum(x) + get_1d_sum(y))
```
