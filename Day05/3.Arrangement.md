```python
# Problem Name : Arrangement (Move Zeroes to End)
# Link : https://www.geeksforgeeks.org/dsa/move-zeroes-end-array/
# Approach : Two Pointer

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    a = [int(x) for x in data[1:n+1]]
    
    # Move zeroes to end in-place
    count = 0
    for i in range(n):
        if a[i] != 0:
            a[count], a[i] = a[i], a[count]
            count += 1
            
    print(" ".join(map(str, a)))

if __name__ == '__main__':
    main()
```
