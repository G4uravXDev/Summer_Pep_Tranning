```python
# Problem Name : Protect the Island
# Link : https://codeforces.com/problemset/problem/1320/C
# Approach : Sweep-Line with Segment Tree (Lazy Propagation)

import sys

# Increase recursion depth for safety
sys.setrecursionlimit(200000)

MAX_VAL = 1000000
tree = [0] * (4 * MAX_VAL + 5)
lazy = [0] * (4 * MAX_VAL + 5)

class Weapon:
    def __init__(self, a, cost):
        self.a = a
        self.cost = cost
    def __lt__(self, other):
        return self.a < other.a

class Monster:
    def __init__(self, x, y, z):
        self.x = x
        self.y = y
        self.z = z
    def __lt__(self, other):
        return self.x < other.x

def build(node, start, end, initial_values):
    if start == end:
        tree[node] = initial_values[start]
        return
    mid = (start + end) >> 1
    build(2 * node, start, mid, initial_values)
    build(2 * node + 1, mid + 1, end, initial_values)
    tree[node] = max(tree[2 * node], tree[2 * node + 1])

def push(node):
    if lazy[node] != 0:
        val = lazy[node]
        lazy[2 * node] += val
        tree[2 * node] += val
        lazy[2 * node + 1] += val
        tree[2 * node + 1] += val
        lazy[node] = 0

def update(node, start, end, l, r, val):
    if l > end or r < start:
        return
    if l <= start and end <= r:
        tree[node] += val
        lazy[node] += val
        return
    push(node)
    mid = (start + end) >> 1
    update(2 * node, start, mid, l, r, val)
    update(2 * node + 1, mid + 1, end, l, r, val)
    tree[node] = max(tree[2 * node], tree[2 * node + 1])

def query(node, start, end, l, r):
    if l > end or r < start:
        return -float('inf')
    if l <= start and end <= r:
        return tree[node]
    push(node)
    mid = (start + end) >> 1
    return max(query(2 * node, start, mid, l, r), query(2 * node + 1, mid + 1, end, l, r))

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    
    n = int(data[0])
    if n == -1:
        return
    m = int(data[1])
    p = int(data[2])
    
    idx = 3
    weapons = []
    for _ in range(n):
        weapons.append(Weapon(int(data[idx]), int(data[idx+1])))
        idx += 2
        
    min_armor_cost = [10**15] * (MAX_VAL + 5)
    for _ in range(m):
        b = int(data[idx])
        cb = int(data[idx+1])
        idx += 2
        if cb < min_armor_cost[b]:
            min_armor_cost[b] = cb
            
    monsters = []
    for _ in range(p):
        monsters.append(Monster(int(data[idx]), int(data[idx+1]), int(data[idx+2])))
        idx += 3
        
    weapons.sort()
    monsters.sort()
    
    initial_values = [-min_armor_cost[i] for i in range(MAX_VAL + 5)]
    build(1, 1, MAX_VAL, initial_values)
    
    ans = -float('inf')
    monster_ptr = 0
    
    for i in range(n):
        while monster_ptr < p and monsters[monster_ptr].x < weapons[i].a:
            y = monsters[monster_ptr].y
            z = monsters[monster_ptr].z
            if y + 1 <= MAX_VAL:
                update(1, 1, MAX_VAL, y + 1, MAX_VAL, z)
            monster_ptr += 1
            
        current_max_armor_profit = query(1, 1, MAX_VAL, 1, MAX_VAL)
        profit = current_max_armor_profit - weapons[i].cost
        if profit > ans:
            ans = profit
            
    print(ans)

if __name__ == '__main__':
    main()
```
