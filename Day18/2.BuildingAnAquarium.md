```python
# Problem Name: Building An Aquarium
# Link : https://codeforces.com/contest/1873/problem/E
# Topic : Binary Search on Answer 

def can_achieve(target_height, A, x):
    cost = 0
    for h in A:
        if h < target_height:
            cost += target_height - h
        if cost > x:
            return False
    return cost <= x

def solve():
    import sys
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    
    idx = 0
    t = int(data[idx])
    idx += 1
    
    results = []
    
    for _ in range(t):
        n = int(data[idx])
        x = int(data[idx+1])
        idx += 2
        
        A = []
        for _ in range(n):
            A.append(int(data[idx]))
            idx += 1
        
        min_h = min(A)
        max_h = max(A)
        
        low = min_h
        high = 2 * 10**9 
        ans = min_h
        
        while low <= high:
            mid = low + (high - low) // 2
            if can_achieve(mid, A, x):
                ans = mid
                low = mid + 1
            else:
                high = mid - 1
        
        results.append(str(ans))
    
    print("\n".join(results))

if __name__ == "__main__":
    solve()
```
