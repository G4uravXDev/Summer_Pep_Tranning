```python
# Problem Name: Combination Sum III
# Link : https://leetcode.com/problems/combination-sum-iii/
# Topic : Recursion and Backtracking


class Solution:
    def combinationSum3(self, k: int, n: int) -> list[list[int]]:
        ans = []
        
        def backtrack(start, remaining_k, current_sum, current_combination):
            if remaining_k == 0 and current_sum == n:
                ans.append(list(current_combination))
                return
            
            if remaining_k == 0 or current_sum > n:
                return
            
            for i in range(start, 10):
                if current_sum + i > n:
                    break
                current_combination.append(i)
                backtrack(i + 1, remaining_k - 1, current_sum + i, current_combination)
                current_combination.pop()
        
        backtrack(1, k, 0, [])
        return ans
```
