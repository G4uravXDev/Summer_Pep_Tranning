```python
# Problem Name: Missing Coin Sum
# Link : https://cses.fi/problemset/task/2183
# Topic : Greedy Approach 

import sys
def solve():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    coins = [int(x) for x in data[1:n+1]]
    coins.sort()

    reachable_sum = 0

    for coin in coins:
        if coin > reachable_sum + 1:
            break
        reachable_sum += coin

    print(reachable_sum + 1)

if __name__ == "__main__":
    solve()

```
