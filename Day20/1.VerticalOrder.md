# Problem Name: Vertical Order Traversal
# Link : https://www.geeksforgeeks.org/problems/print-vertical-order-of-a-binary-tree/1
# Topic : Tree Traversal


class Node:
    def __init__(self, data):
        self.data = data
        self.left = None
        self.right = None

class Solution:
    def verticalOrder(self, root):
        if root is None:
            return []

        hd_map = {}
        

        queue = [(root, 0)]
        
        min_hd = 0
        max_hd = 0

        while queue:
            node, hd = queue.pop(0)

            if hd not in hd_map:
                hd_map[hd] = []
            hd_map[hd].append(node.data)

            min_hd = min(min_hd, hd)
            max_hd = max(max_hd, hd)

            if node.left:
                queue.append((node.left, hd - 1))

            if node.right:
                queue.append((node.right, hd + 1))

        result = []
        for hd in range(min_hd, max_hd + 1):
            if hd in hd_map:
                result.extend(hd_map[hd])

        return result
