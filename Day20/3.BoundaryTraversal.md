# Problem Name: Boundary Traversal Of Binary Tree
# Link :  https://www.geeksforgeeks.org/problems/boundary-traversal-of-binary-tree/1
# Topic : Tree Traversal

class Node:
    def __init__(self, data):
        self.data = data
        self.left = None
        self.right = None

class Solution:
    def boundaryTraversal(self, root):
        if not root:
            return []

        result = [root.data]

        if not root.left and not root.right:
            return result

        curr = root.left
        while curr:
            if curr.left or curr.right:
                result.append(curr.data)
            if curr.left:
                curr = curr.left
            else:
                curr = curr.right

        self.addLeaves(root, result)

        stack = []
        curr = root.right
        while curr:
            if curr.left or curr.right:
                stack.append(curr.data)
            if curr.right:
                curr = curr.right
            else:
                curr = curr.left

        while stack:
            result.append(stack.pop())

        return result

    def addLeaves(self, node, result):
        if not node:
            return

        self.addLeaves(node.left, result)

        if not node.left and not node.right:
            result.append(node.data)

        self.addLeaves(node.right, result)
