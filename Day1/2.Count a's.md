/**
 * Problem Name : Count a's
 * Link : https://www.geeksforgeeks.org/count-occurrences-of-a-specific-character-in-all-substrings-of-a-given-string/
 * Approach : Prefix Sum Technique
 */

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int q = sc.nextInt();
        String s = sc.next();

        int[] pref = new int[n + 1];

        for (int i = 1; i <= n; i++) {
            pref[i] = pref[i - 1] + (s.charAt(i - 1) == 'a' ? 1 : 0);
        }

        StringBuilder sb = new StringBuilder();

        while (q-- > 0) {
            int l = sc.nextInt();
            int r = sc.nextInt();
            sb.append(pref[r] - pref[l - 1]).append("\n");
        }

        System.out.print(sb);
    }
}
