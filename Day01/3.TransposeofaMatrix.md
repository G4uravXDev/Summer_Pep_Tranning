```python
# Problem Name : Transpose Matrix
# Link : https://leetcode.com/problems/transpose-matrix/
# Approach : In-Place Matrix Transpose (Diagonal Swapping)

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    
    mat = []
    idx = 1
    for i in range(n):
        row = []
        for j in range(n):
            row.append(int(data[idx]))
            idx += 1
        mat.append(row)
        
    for i in range(n):
        for j in range(i + 1, n):
            mat[i][j], mat[j][i] = mat[j][i], mat[i][j]
            
    out = []
    for i in range(n):
        out.append(" ".join(map(str, mat[i])))
    print("\n".join(out))

if __name__ == '__main__':
    main()
```
