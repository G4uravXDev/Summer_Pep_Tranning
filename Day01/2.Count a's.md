# Problem Name : Count a's
# Link : https://www.geeksforgeeks.org/count-occurrences-of-a-specific-character-in-all-substrings-of-a-given-string/
# Approach : Prefix Sum Technique

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    q = int(data[1])
    s = data[2]
    
    pref = [0] * (n + 1)
    for i in range(1, n + 1):
        pref[i] = pref[i - 1] + (1 if s[i - 1] == 'a' else 0)
        
    out = []
    idx = 3
    for _ in range(q):
        l = int(data[idx])
        r = int(data[idx+1])
        idx += 2
        out.append(str(pref[r] - pref[l - 1]))
        
    print("\n".join(out))

if __name__ == '__main__':
    main()
