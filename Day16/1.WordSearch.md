```python
# Problem Name: Word Search
# Link : https://leetcode.com/problems/word-search/description/
# Topic : Recursion & Backtracking

class Solution:
    def exist(self, board: list[list[str]], word: str) -> bool:
        m = len(board)
        n = len(board[0])
        visited = [[False] * n for _ in range(m)]

        def solve(row, col, idx):
            if idx == len(word):
                return True

            dx = [0, 1, 0, -1]
            dy = [1, 0, -1, 0]

            for i in range(4):
                newX = row + dx[i]
                newY = col + dy[i]

                if 0 <= newX < m and 0 <= newY < n and not visited[newX][newY] and board[newX][newY] == word[idx]:
                    visited[newX][newY] = True
                    if solve(newX, newY, idx + 1):
                        return True
                    visited[newX][newY] = False

            return False

        for i in range(m):
            for j in range(n):
                if board[i][j] == word[0]:
                    visited[i][j] = True
                    if solve(i, j, 1):
                        return True
                    visited[i][j] = False

        return False
```
