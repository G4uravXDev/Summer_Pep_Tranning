# Problem Name: N-Queens
# Link : https://leetcode.com/problems/n-queens/
# Topic : Recursion & Backtracking

class Solution:
    def solveNQueens(self, n: int) -> list[list[str]]:
        result = []
        current_board = []
        
        def isSafe(row, col):
            for i in range(row):
                if current_board[i][col] == 'Q':
                    return False
            
            i, j = row - 1, col - 1
            while i >= 0 and j >= 0:
                if current_board[i][j] == 'Q':
                    return False
                i -= 1
                j -= 1
                
            i, j = row - 1, col + 1
            while i >= 0 and j < n:
                if current_board[i][j] == 'Q':
                    return False
                i -= 1
                j += 1
                
            return True

        def solve(row):
            if row == n:
                result.append(list(current_board))
                return

            for col in range(n):
                if isSafe(row, col):
                    row_str = ["."] * n
                    row_str[col] = "Q"
                    current_board.append("".join(row_str))
                    
                    solve(row + 1)
                    
                    current_board.pop()

        solve(0)
        return result
