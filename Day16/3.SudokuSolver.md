# Problem Name: Sudoku Solver
# Link : https://leetcode.com/problems/sudoku-solver/
# Topic : Recursion & Backtracking

class Solution:
    def solveSudoku(self, board: list[list[str]]) -> None:
        self.solve(board)

    def solve(self, board: list[list[str]]) -> bool:
        for i in range(9):
            for j in range(9):
                if board[i][j] == '.':
                    for c in map(str, range(1, 10)):
                        if self.isSafe(board, i, j, c):
                            board[i][j] = c
                            if self.solve(board):
                                return True
                            board[i][j] = '.'
                    return False
        return True

    def isSafe(self, board: list[list[str]], row: int, col: int, c: str) -> bool:
        for j in range(9):
            if board[row][j] == c:
                return False

        for i in range(9):
            if board[i][col] == c:
                return False

        boxRow = (row // 3) * 3
        boxCol = (col // 3) * 3

        for i in range(boxRow, boxRow + 3):
            for j in range(boxCol, boxCol + 3):
                if board[i][j] == c:
                    return False

        return True
