```python
# Problem Name: Rat in a Maze
# Link : https://www.geeksforgeeks.org/problems/rat-in-a-maze-problem/1
# Topic : Recursion & Backtracking

class Solution:
    def ratInMaze(self, maze: list[list[int]]) -> list[str]:
        n = len(maze)
        ans = []
        if maze[0][0] == 0:
            return ans

        visited = [[False] * n for _ in range(n)]
        
        def solve(row, col, path):
            if row == n - 1 and col == n - 1:
                ans.append(path)
                return
                
            dx = [1, 0, 0, -1]
            dy = [0, -1, 1, 0]
            dir_char = ['D', 'L', 'R', 'U']

            for i in range(4):
                newX = row + dx[i]
                newY = col + dy[i]

                if 0 <= newX < n and 0 <= newY < n and not visited[newX][newY] and maze[newX][newY] == 1:
                    visited[newX][newY] = True
                    solve(newX, newY, path + dir_char[i])
                    visited[newX][newY] = False
                    
        visited[0][0] = True
        solve(0, 0, "")
        return ans
```
