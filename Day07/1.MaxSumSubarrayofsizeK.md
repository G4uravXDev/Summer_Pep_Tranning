```python
# Problem Name : MaxSumSubarrayofsizeK
# Link : https://www.geeksforgeeks.org/problems/max-sum-subarray-of-size-k5313/1
# Approach : Sliding Window

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    k = int(data[1])
    a = [int(x) for x in data[2:n+2]]
    
    current_sum = sum(a[:k])
    max_sum = current_sum
    
    for i in range(k, n):
        current_sum += a[i] - a[i - k]
        max_sum = max(max_sum, current_sum)
        
    print(max_sum)

if __name__ == '__main__':
    main()
```
