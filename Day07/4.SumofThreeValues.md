```python
# Problem Name : Sum of Three Values
# Link : https://cses.fi/problemset/task/1641/
# Approach : Sorting & Two Pointers (O(N^2))

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    x = int(data[1])
    a = [int(v) for v in data[2:n+2]]
    
    pairs = sorted([(a[i], i) for i in range(n)])
    
    for i in range(n - 2):
        first = pairs[i][0]
        target = x - first
        left = i + 1
        right = n - 1
        
        while left < right:
            current_sum = pairs[left][0] + pairs[right][0]
            if current_sum == target:
                print(f"{pairs[i][1] + 1} {pairs[left][1] + 1} {pairs[right][1] + 1}")
                return
            elif current_sum < target:
                left += 1
            else:
                right -= 1
                
    print("IMPOSSIBLE")

if __name__ == '__main__':
    main()
```
