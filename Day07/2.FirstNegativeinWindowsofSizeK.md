# Problem Name : First Negative in Window of Size K
# Link : https://www.geeksforgeeks.org/problems/first-negative-integer-in-every-window-of-size-k3345/1
# Approach : Sliding Window with Deque

import sys
from collections import deque

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    k = int(data[1])
    a = [int(x) for x in data[2:n+2]]
    
    dq = deque()
    ans = []
    
    for i in range(n):
        if dq and dq[0] < i - k + 1:
            dq.popleft()
            
        if a[i] < 0:
            dq.append(i)
            
        if i >= k - 1:
            if dq:
                ans.append(a[dq[0]])
            else:
                ans.append(0)
                
    print(" ".join(map(str, ans)))

if __name__ == '__main__':
    main()
