```python
# Problem Name : KSizedSubarrayMaximum
# Link : https://www.geeksforgeeks.org/problems/maximum-of-all-subarrays-of-size-k3101/1
# Approach : Sliding Window (using Monotonic Deque)

import sys
from collections import deque

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    k = int(data[1])
    a = [int(x) for x in data[2:n+2]]
    
    dq = deque()
    ans = []
    
    for i in range(n):
        while dq and dq[0] < i - k + 1:
            dq.popleft()
            
        while dq and a[dq[-1]] <= a[i]:
            dq.pop()
            
        dq.append(i)
        
        if i >= k - 1:
            ans.append(a[dq[0]])
            
    print(" ".join(map(str, ans)))

if __name__ == '__main__':
    main()
```
