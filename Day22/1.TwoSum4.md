# Problem Name: Two Sum IV - Input is a BST
# Link : https://leetcode.com/problems/two-sum-iv-input-is-a-bst/description/
# Topic : Binary Search Tree


class Solution:
    def findTarget(self, root: Optional[TreeNode], k: int) -> bool:
        if not root:
            return False
        
        set_nodes = set()
        
        q=deque()
        q.append(root)

        while q:
            node = q.popleft()
            
            if k-node.val in set_nodes:
                return True
            
            set_nodes.add(node.val)
            
            if node.left:
                q.append(node.left)
            
            if node.right:
                q.append(node.right)
                
        return False
