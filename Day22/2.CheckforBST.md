```python
# Problem Name: Check for BST
# Link : https://leetcode.com/problems/two-sum-iv-input-is-a-bst/description/
# Topic : Binary Search Tree 


class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        if not root:
            return True
        
        q=deque()
        q.append(root)

        while q:
            node = q.popleft()
            
            if node.left:
                if node.left.val >= node.val:
                    return False
                q.append(node.left)
            
            if node.right:
                if node.right.val <= node.val:
                    return False
                q.append(node.right)
                
        return True
```
