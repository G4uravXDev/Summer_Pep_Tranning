```python
# Problem Name: Graph Adjacency List Traversal
# Link : https://www.geeksforgeeks.org/problems/print-adjacency-list-1587115620/1
# Topic : Graph Adjacency List 


class Solution:
    def printAdjList(self, V, adj):
        adj_list = defaultdict(list)
        
        for u, v in adj:
            adj_list[u].append(v)
            adj_list[v].append(u)
        
        return adj_list
```
