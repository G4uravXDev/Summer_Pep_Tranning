# Problem Name: Count Good Numbers
# Link : https://leetcode.com/problems/count-good-numbers/
# Topic : Recursion

class Solution:
    def countGoodNumbers(self, n: int) -> int:
        MOD = 1_000_000_007
        
        def expo(a, b):
            if b == 0:
                return 1
            if b % 2 == 1:
                return (a * expo((a * a) % MOD, b // 2)) % MOD
            return expo((a * a) % MOD, b // 2)

        odd_index = n // 2
        even_index = (n + 1) // 2

        ans = (expo(5, even_index) * expo(4, odd_index)) % MOD
        return ans
