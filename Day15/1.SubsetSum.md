```python
# Problem Name: Subset Sum Problem
# Link : https://www.geeksforgeeks.org/problems/subset-sum-problem-1611555638/1
# Topic : Recursion

class Solution:
    def isSubsetSum(self, arr: list[int], sum: int) -> bool:
        def solve(idx, currSum):
            if currSum == sum:
                return True
            if idx == len(arr) or currSum > sum:
                return False
            
            # option 1: pick
            pick = solve(idx + 1, currSum + arr[idx])
            if pick:
                return True
            return solve(idx + 1, currSum)

        return solve(0, 0)
```
