# Problem Name: Subset Sum Problem
# Link : https://www.geeksforgeeks.org/problems/subset-sum-problem-1611555638/1
# Topic : Recursion

class Solution:
    def solve(self, idx: int, currSum: int, target: int, arr: list[int]) -> bool:
        if currSum == target:
            return True
        if idx == len(arr) or currSum > target:
            return False

        pick = self.solve(idx + 1, currSum + arr[idx], target, arr)
        if pick:
            return True
        notPick = self.solve(idx + 1, currSum, target, arr)
        return notPick

    def isSubsetSum(self, arr: list[int], sum: int) -> bool:
        return self.solve(0, 0, sum, arr)
