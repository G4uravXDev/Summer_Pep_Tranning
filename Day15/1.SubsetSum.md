/**
* Problem Name: Subset Sum Problem
* Link : https://www.geeksforgeeks.org/problems/subset-sum-problem-1611555638/1
* Topic : Recursion
*/
class Solution {
    public boolean solve(int idx, int currSum, int target, int[] arr) {
        if (currSum == target)
            return true;
        if (idx == arr.length || currSum > target) {
            return false;
        }

        boolean pick = solve(idx + 1, currSum + arr[idx], target, arr);
        if (pick)
            return true;
        boolean notPick = solve(idx + 1, currSum, target, arr);
        return notPick;
    }

    public boolean isSubsetSum(int[] arr, int sum) {
        return solve(0, 0, sum, arr);
    }
}
