```python
# Problem Name: Subsets II
# Link : https://leetcode.com/problems/subsets-ii/description/
# Topic : Recursion

class Solution:
    def solve(self, idx: int, curr: list[int], ans: list[list[int]], nums: list[int]) -> None:
        ans.append(list(curr))

        for i in range(idx, len(nums)):
            if i > idx and nums[i] == nums[i - 1]:
                continue
            curr.append(nums[i])
            self.solve(i + 1, curr, ans, nums)
            curr.pop()

    def subsetsWithDup(self, nums: list[int]) -> list[list[int]]:
        nums.sort()
        curr = []
        ans = []
        self.solve(0, curr, ans, nums)
        return ans
```
