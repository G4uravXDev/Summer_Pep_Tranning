```python
# Problem Name : Overlapping Intervals
# Link : https://www.geeksforgeeks.org/problems/overlapping-intervals--170633/1
# Approach : Sorting

class Solution:
    def mergeOverlap(self, arr: list[list[int]]) -> list[list[int]]:
        arr.sort(key=lambda x: x[0])
        ans = []
        
        for interval in arr:
            if not ans or ans[-1][1] < interval[0]:
                ans.append(list(interval))
            else:
                ans[-1][1] = max(ans[-1][1], interval[1])
                
        return ans
```
