```python
# Problem Name : Tower
# Link : https://cses.fi/problemset/task/1073
# Approach : Greedy

import sys
import bisect

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    
    towers = []
    for i in range(1, n + 1):
        x = int(data[i])
        idx = bisect.bisect_right(towers, x)
        if idx == len(towers):
            towers.append(x)
        else:
            towers[idx] = x
            
    print(len(towers))

if __name__ == '__main__':
    main()
```
