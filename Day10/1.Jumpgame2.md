/**
 * Problem Name : Jump Game II
 * Link : https://leetcode.com/problems/jump-game-ii/description/
 * Approach : Greedy
 */


class Solution {
    public int jump(int[] nums) {
        int n = nums.length;
        if (n <= 1) return 0;

        int jumps = 0;
        int currentReach = 0; 
        int maxReach = 0;    

        for (int i = 0; i < n - 1; i++) {
            maxReach = Math.max(maxReach, i + nums[i]);

            if (i == currentReach) {
                jumps++;
                currentReach = maxReach;
            }
        }

        return jumps;
    }
}
