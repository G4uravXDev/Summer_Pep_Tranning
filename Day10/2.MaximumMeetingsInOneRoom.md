```python
# Problem Name : Maximum Meetings In One Room
# Link : https://www.geeksforgeeks.org/problems/maximum-meetings-in-one-room/1
# Approach : Greedy Algorithm

class Solution:
    def maxMeetings(self, start: list[int], end: list[int], N: int) -> list[int]:
        meetings = []
        for i in range(N):
            meetings.append((end[i], start[i], i + 1))
            
        meetings.sort(key=lambda x: (x[0], x[2]))
        
        result = []
        last_finish_time = -1
        
        for end_time, start_time, pos in meetings:
            if start_time > last_finish_time:
                result.append(pos)
                last_finish_time = end_time
                
        result.sort()
        return result
```
