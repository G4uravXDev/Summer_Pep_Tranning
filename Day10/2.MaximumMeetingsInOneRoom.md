/**
 * Problem Name : Maximum Meetings In One Room
 * Link : https://www.geeksforgeeks.org/problems/maximum-meetings-in-one-room/1
 * Approach : Greedy Algorithm
 */


import java.util.*;

class Solution {
    class Pair implements Comparable<Pair> {
        int start, end, pos;

        Pair(int start, int end, int pos) {
            this.start = start;
            this.end = end;
            this.pos = pos;
        }

        @Override
        public int compareTo(Pair other) {
            if (this.end != other.end) {
                return Integer.compare(this.end, other.end);
            }
            return Integer.compare(this.pos, other.pos);
        }
    }

    public ArrayList<Integer> maxMeetings(int[] start, int[] end, int N) {
        ArrayList<Pair> list = new ArrayList<>();
        for (int i = 0; i < N; i++) {
            list.add(new Pair(start[i], end[i], i + 1));
        }

        Collections.sort(list);

        ArrayList<Integer> result = new ArrayList<>();
        int lastFinishTime = -1;

        for (Pair p : list) {
            if (p.start > lastFinishTime) {
                result.add(p.pos);
                lastFinishTime = p.end;
            }
        }

        return result;
    }
}
