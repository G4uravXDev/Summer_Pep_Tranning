```python
# Problem Name : Minimum Platforms
# Link : https://www.geeksforgeeks.org/problems/minimum-platforms-1587115620/1
# Approach : Greedy Algorithm

class Solution:
    def findPlatform(self, arr: list[int], dep: list[int], n: int) -> int:
        arr.sort()
        dep.sort()

        platforms = 1     
        max_platforms = 1
        i = 1             
        j = 0            

        while i < n and j < n:
            if arr[i] <= dep[j]:
                platforms += 1
                i += 1
            else:
                platforms -= 1
                j += 1

            max_platforms = max(max_platforms, platforms)

        return max_platforms
```
