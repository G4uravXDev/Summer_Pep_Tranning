```python
# Problem Name : Distinct Characters in a Window
# Link : https://www.naukri.com/code360/problems/distinct-characters_2221410
# Approach : Sliding Window

import sys

def solve(s, k):
    n = len(s)
    if n == 0 or k == 0:
        return 0

    freq = {}
    left = 0
    max_len = 0

    for right in range(n):
        c = s[right]
        freq[c] = freq.get(c, 0) + 1

        while len(freq) > k:
            left_char = s[left]
            freq[left_char] -= 1
            if freq[left_char] == 0:
                del freq[left_char]
            left += 1

        max_len = max(max_len, right - left + 1)
    return max_len

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    t = int(data[0])
    
    idx = 1
    for _ in range(t):
        if idx >= len(data):
            break
        k = int(data[idx])
        s = data[idx+1]
        idx += 2
        print(solve(s, k))

if __name__ == '__main__':
    main()
```
