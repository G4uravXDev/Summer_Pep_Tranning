```python
# Problem Name : Subarrays with K Different Integers
# Link : https://leetcode.com/problems/subarrays-with-k-different-integers/description/
# Approach : Sliding Window

import sys

def atMostK(nums, k):
    if k < 0:
        return 0

    n = len(nums)
    freq = {}
    left = 0
    count = 0

    for right in range(n):
        freq[nums[right]] = freq.get(nums[right], 0) + 1

        while len(freq) > k:
            freq[nums[left]] -= 1
            if freq[nums[left]] == 0:
                del freq[nums[left]]
            left += 1

        count += (right - left + 1)

    return count

def subarraysWithKDistinct(nums, k):
    return atMostK(nums, k) - atMostK(nums, k - 1)

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    k = int(data[1])
    a = [int(x) for x in data[2:n+2]]
    print(subarraysWithKDistinct(a, k))

if __name__ == '__main__':
    main()
```
