```python
# Problem Name : Previous Greater Element
# Link : https://www.geeksforgeeks.org/problems/previous-greater-element/1
# Approach : Stack

import sys

def solve(a, n):
    result = [-1] * n
    stack = []

    for i in range(n):
        while stack and stack[-1] <= a[i]:
            stack.pop()
        if stack:
            result[i] = stack[-1]
        stack.append(a[i])
    return result

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    t = int(data[0])
    
    idx = 1
    for _ in range(t):
        if idx >= len(data):
            break
        n = int(data[idx])
        a = [int(x) for x in data[idx+1 : idx+1+n]]
        idx += 1 + n
        result = solve(a, n)
        print(result)

if __name__ == '__main__':
    main()
```
