/**
 * Problem Name : 1652. Forest Queries
 * Link : https://cses.fi/problemset/task/1652
 * Approach : Prefix Sum Technique
 */


import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int q = sc.nextInt();

        String[] arr = new String[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.next();
        }

        int[][] prefix = new int[n][n];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                prefix[i][j] = (arr[i].charAt(j) == '*') ? 1 : 0;

                if (i > 0) 
                    prefix[i][j] += prefix[i - 1][j];
                if (j > 0) 
                    prefix[i][j] += prefix[i][j - 1];
                if (i > 0 && j > 0) 
                    prefix[i][j] -= prefix[i - 1][j - 1];
            }
        }

        while (q-- > 0) {
            int y1 = sc.nextInt();
            int x1 = sc.nextInt();
            int y2 = sc.nextInt();
            int x2 = sc.nextInt();

            y1--; x1--; y2--; x2--;

            int ans = prefix[y2][x2];

            if (y1 > 0) 
                ans -= prefix[y1 - 1][x2];
            if (x1 > 0) 
                ans -= prefix[y2][x1 - 1];
            if (y1 > 0 && x1 > 0) 
                ans += prefix[y1 - 1][x1 - 1];

            System.out.println(ans);
        }
    }
}
