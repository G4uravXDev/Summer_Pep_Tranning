# Problem Name : Forest Queries
# Link : https://cses.fi/problemset/task/1652
# Approach : Prefix Sum Technique

import sys

def main():
    input = sys.stdin.read
    data = input().split()
    if not data:
        return
    n = int(data[0])
    q = int(data[1])
    
    arr = data[2:n+2]
    
    prefix = [[0] * n for _ in range(n)]
    
    for i in range(n):
        for j in range(n):
            prefix[i][j] = 1 if arr[i][j] == '*' else 0
            if i > 0:
                prefix[i][j] += prefix[i - 1][j]
            if j > 0:
                prefix[i][j] += prefix[i][j - 1]
            if i > 0 and j > 0:
                prefix[i][j] -= prefix[i - 1][j - 1]
                
    out = []
    idx = n + 2
    for _ in range(q):
        y1 = int(data[idx])
        x1 = int(data[idx+1])
        y2 = int(data[idx+2])
        x2 = int(data[idx+3])
        idx += 4
        
        y1 -= 1
        x1 -= 1
        y2 -= 1
        x2 -= 1
        
        ans = prefix[y2][x2]
        if y1 > 0:
            ans -= prefix[y1 - 1][x2]
        if x1 > 0:
            ans -= prefix[y2][x1 - 1]
        if y1 > 0 and x1 > 0:
            ans += prefix[y1 - 1][x1 - 1]
        out.append(str(ans))
        
    print("\n".join(out))

if __name__ == '__main__':
    main()
