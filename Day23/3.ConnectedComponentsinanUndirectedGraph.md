# Problem Name: Connected Components in an Undirected Graph
# Link : https://www.geeksforgeeks.org/problems/connected-components-in-an-undirected-graph/1
# Topic : Graph Traversal (Number of Connected Components)


class Solution:
    def countComponents(self, V : int, adj : List[List[int]]) -> int:
        
        visited = [False] * V
        count = 0
        for i in range(V):
            if not visited[i]:
                self.dfs(i, visited, adj)
                count += 1
        return count
    
    def dfs(self, node, visited, adj):
        visited[node] = True
        for neighbor in adj[node]:
            if not visited[neighbor]:
                self.dfs(neighbor, visited, adj)
