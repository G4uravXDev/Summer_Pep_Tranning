# Problem Name: DFS of Graph
# Link : https://www.geeksforgeeks.org/problems/depth-first-traversal-for-a-graph/1
# Topic : Graph Traversal (DFS)

from typing import List


class Solution:
    def dfsOfGraph(self, adj: List[List[int]]) -> List[int]:
        V = len(adj)
        visited = [False] * V
        result = []
        
        self.dfs(0, visited, result, adj)
        return result
    
    def dfs(self, node: int, visited: List[bool], result: List[int], adj: List[List[int]]):
        visited[node] = True
        result.append(node)
        
        for neighbor in adj[node]:
            if not visited[neighbor]:
                self.dfs(neighbor, visited, result, adj)
