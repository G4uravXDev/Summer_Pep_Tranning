# Problem Name: Number of Provinces (Connected Components)
# Link : https://www.geeksforgeeks.org/problems/number-of-provinces/1
# Topic : Graph Traversal (Number of Connected Components)

from typing import List


class Solution:
    def numProvinces(self, adj: List[List[int]], V: int = None) -> int:
        n = V if V is not None else len(adj)
        visited = [False] * n
        count = 0
        for i in range(n):
            if not visited[i]:
                self.dfs(i, visited, adj, n)
                count += 1
        return count
    
    def dfs(self, node: int, visited: List[bool], adj: List[List[int]], n: int):
        visited[node] = True
        for neighbor in range(n):
            if adj[node][neighbor] == 1 and not visited[neighbor]:
                self.dfs(neighbor, visited, adj, n)
