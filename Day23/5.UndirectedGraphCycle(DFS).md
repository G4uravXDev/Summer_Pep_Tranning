# Problem Name: Detect Cycle in an Undirected Graph
# Link : https://www.geeksforgeeks.org/problems/detect-cycle-in-an-undirected-graph/1
# Topic : Cycle in an Undirected Graph(DFS)

from typing import List


class Solution:
    def isCycle(self, V: int = None, adj: List[List[int]] = None) -> bool:
        if adj is None:
            adj = V
            V = len(adj)
        elif V is None:
            V = len(adj)
            
        visited = [False] * V
        for i in range(V):
            if not visited[i]:
                if self.dfs(i, -1, visited, adj):
                    return True
        return False
    
    def dfs(self, node: int, parent: int, visited: List[bool], adj: List[List[int]]) -> bool:
        visited[node] = True
        for neighbor in adj[node]:
            if not visited[neighbor]:
                if self.dfs(neighbor, node, visited, adj):
                    return True
            elif neighbor != parent:
                return True
        return False
